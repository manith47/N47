Promise.all([d3.json("sgmap.json"), d3.csv("population2022.csv")]).then(data => {

// control that shows state info on hover
const info = L.control();

this._div = L.DomUtil.create('div', 'info');
this.update();
return this._div;
};

info.update = function (props) {
const contents = props ? `<b>${props.Name}</b><br />${d3.format(",")(props.popdata)} people` : 'Hover over a subdistrict';
this._div.innerHTML = `<h4>SG Population</h4>${contents}`;

// get color depending on population density value
function getColor(d) {
return d > 35000 ? '#800026' :
d > 30000 ? '#BD0026' :
d > 25000 ? '#E31A1C' :
d > 20000 ? '#FC4E2A' :
d > 15000 ? '#FD8D3C' :
d > 10000 ? '#FEB24C' :
d > 5000 ? '#FED976' :
d > 10 ? '#FFEDA0' : '#333';
}

function style(feature) {
return {
weight: 1,
opacity: 1,
color: 'grey',
fillOpacity: 0.7,
fillColor: getColor(feature.properties.popdata)
};
}

function highlightFeature(e) {
const layer = e.target;

layer.setStyle({
weight: 5,
color: '#666',
dashArray: '',
fillOpacity: 0.7
});

layer.bringToFront();

info.update(layer.feature.properties);
}

/* global statesData */
const geojson = L.geoJson(mapData, {
style,
onEachFeature
}).addTo(map);

function resetHighlight(e) {
geojson.resetStyle(e.target);
info.update();
}

function zoomToFeature(e) {
map.fitBounds(e.target.getBounds());
}

function onEachFeature(feature, layer) {
layer.on({
mouseover: highlightFeature,
mouseout: resetHighlight,
click: zoomToFeature
});
}

const legend = L.control({position: 'bottomright'});

legend.onAdd = function (map) {

const div = L.DomUtil.create('div', 'info legend');

const labels = [];
let from, to;

for (let i = 0; i < grades.length; i++) {
from = grades[i];
to = grades[i + 1];

labels.push(`<i style="background:${getColor(from + 1)}"></i> ${from}${to ? `&ndash;${to}` : '+'}`);
}

div.innerHTML = labels.join('<br>');
return div;
};

legend.addTo(map);
})
